                points.add(new CornerPoint(initialLat +  Min + (Math.random() * ((Max - Min) )),

            }

            if(j % (int) Math.sqrt(numberPolygons) == 0){
                initialLon += 0.003;
                initialLat = GlobalConstants.lastLocationOnMap.getLatitude();

            initialLat += 0.003;


            polis.add(p);
        }
